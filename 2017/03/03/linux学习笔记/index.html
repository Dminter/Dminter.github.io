ls

ls 列出目录文件或者文件夹
ls -l 详细信息
ls -a 更详细信息，包含隐藏文件

mkdir

mkdir 创建文件夹
mkdir -p /aa/bb 创建bb文件夹若没有aa则创建aa

cd

cd 切换目录
cd .. 回到上层目录
cd dev回车 可以补全dev文件夹名称

touch

touch 1.txt 创建文件 1.txt

cat

cat 2.txt 显示2.txt文件内容

cp

cp 2.txt 2 复制2.txt到2文件夹里面

apt

apt install 安装软件
apt install subversion 安装SVN

mv

mv 2.txt  3.txt 重命名
mv 3.txt 2 把3.txt移动到2文件夹里面

rm

rm x.txt 删除文件x.txt
rm -r 递归删除
rm -f 强制删除
